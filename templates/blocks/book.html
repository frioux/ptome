[%# Takes isbn as an argument also, if id is defined id will be shown, also takes maxlength of title and author %] [%- USE String -%] [%- bookinfo = tome.book_info(isbn) -%]
[%- IF (id) -%]
Book ID:
[% id | html %]
[%- END -%]
ISBN:
[% bookinfo.isbn | html %]
[%- title = String.new(bookinfo.title) -%] [% author = String.new(bookinfo.author) %] [%- DEFAULT maxlength = 30 -%]
Title:
[% title.truncate(maxlength, '~').replace('~', '…') %]
Author:
[% author.truncate(maxlength, '~').replace('~', '…') %]
Edition:
[% bookinfo.edition | html %]
